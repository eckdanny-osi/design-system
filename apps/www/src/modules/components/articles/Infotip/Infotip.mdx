    <CardBody>
      <CardTitle>
        Something Ambiguous

```jsx
import { Infotip } from "@cwds/components";
```

### Uncontrolled Infotip

better

```jsx
import { UncontrolledInfotip } from "@cwds/components";
```

outcome is

<DemoCard>
  <UncontrolledInfotip id="my-uncontrolled-infotip" placement="right">
    <PopoverBody>
      <p className="mb-0">
        The quick brown fox jumps over the lazy dog. The quick brown fox jumps
        over the lazy dog.
      </p>
    </PopoverBody>
  </UncontrolledInfotip>
</DemoCard>

and usage is

```jsx
<UncontrolledInfotip id="my-uncontrolled-infotip" placement="right">
  <PopoverBody>
    <p>
      The quick brown fox jumps over the lazy dog. The quick brown fox jumps
      over the lazy dog.
    </p>
  </PopoverBody>
</UncontrolledInfotip>
```
